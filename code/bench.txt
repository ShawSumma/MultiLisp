(define clock (timer))
(define (bench t) (bench-n 0 t))
(define (countdown n)
    (if (> n 0)
        (countdown (- n 1))))
(define (bench-n x t)
    (countdown 10000)
    (if (< (clock) t)
        (bench-n (+ x 1) t)
        x))
(define (bench-all-n n)
    (print "Score : ")
    (println (bench n))
    (bench-all-n (* n 1.05)))
(define (bench-all)
    (bench-all-n 2))
(bench-all)
